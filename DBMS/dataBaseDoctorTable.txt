CREATE TABLE Doctor(
    did int AUTO_INCREMENT PRIMARY KEY,
    drfirstname VARCHAR(50) NOT NULL,
    drlasttname VARCHAR(50) NOT NULL
);

CREATE TABLE Patient(
    pid int AUTO_INCREMENT PRIMARY KEY,
    pfirstname VARCHAR(50) NOT NULL,
    plasttname VARCHAR(50) NOT NULL,
    insuranceNumber VARCHAR NOT NULL UNIQUE
);

CREATE TABLE Specialization(
    did int NOT NULL,
    description  VARCHAR(100),
    FOREIGN KEY (pid) REFERNCES Doctor(did)
);

CREATE TABLE Diagonse(
    pid int NOT NULL,
    details  VARCHAR(100),
    FOREIGN KEY (pid) REFERNCES Patient(did)
);
